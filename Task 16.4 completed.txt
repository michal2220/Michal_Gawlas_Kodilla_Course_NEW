SELECT I.SUMMARY, IL.NAME
FROM ISSUES I JOIN ISSUESLISTS IL
ON I.ISSUESLIST_ID = IL.ID;

SELECT I.SUMMARY, U.ID AS USER_ID, U.FIRSTNAME, U.LASTNAME
FROM ISSUES I JOIN USERS U
ON I.USER_ID_ASSIGNEDTO = U.ID;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(SUMMARY) AS QUANTITY_OF_TASKS_ASSIGNED
FROM USERS U  JOIN ISSUES I
ON U.ID = I.USER_ID_ASSIGNEDTO
GROUP BY U.FIRSTNAME, U.LASTNAME
HAVING COUNT(*)>0;
